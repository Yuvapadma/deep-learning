from sklearn.datasets import fetch_lfw_people
import matplotlib.pyplot as plt

lfw_people = fetch_lfw_people(min_faces_per_person=70, resize=0.4)

print("Number of samples:", lfw_people.images.shape[0])
print("Image shape:", lfw_people.images[0].shape)
print("Number of classes:", len(lfw_people.target_names))

# Show an example face
plt.imshow(lfw_people.images[0], cmap='gray')
plt.title(lfw_people.target_names[lfw_people.target[0]])
plt.show()
